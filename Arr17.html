class Arr17{
public static void main(String args[]){
int array[] = {10,121,67,34,59,10,67};
System.out.println("Original Array elements after sorting:");  
for (int i = 0; i < array.length; i++){  
for (int j = i + 1; j < array.length; j++)   {  
int tmp = 0;  
if (array[i] > array[j]){  
tmp = array[i];  
array[i] = array[j];  
array[j] = tmp;  
}}    
System.out.print(array[i]+" ");  
}          
int[] temp = new int[array.length];
int newSize = 0;
for (int i = 0; i < array.length; i++){
boolean isDuplicate = false;
for (int j = 0; j < newSize; j++) {
if (array[i] == temp[j]) {
isDuplicate = true;
break;
}}
if(!isDuplicate){
temp[newSize] = array[i];
newSize++;
}}
int[] uniqueArray = new int[newSize];
for (int i = 0; i < newSize; i++) {
uniqueArray[i] = temp[i];
}
System.out.print("\nArray without duplicates: \n");
for (int i = 0; i < uniqueArray.length; i++) {
System.out.print(uniqueArray[i] + " ");
}
System.out.println("\nSecond largest number is : "+uniqueArray[(uniqueArray.length)-2]);
}}
