class BPr47 {
    public static void main(String[] args) {
        double principal = 1000.0; // initial principal amount
        double rate = 5.0; // rate of interest (in %)
        int time = 5; // time period (in years)
        int compoundedPerYear = 1; // number of times interest is compounded per year

        double amount = calculateCompoundInterest(principal, rate, time, compoundedPerYear);

        System.out.printf("The amount after %d years is: %.2f%n", time, amount);
    }

    public static double calculateCompoundInterest(double principal, double rate, int time, int compoundedPerYear) {
        double amount = principal;
        for (int i = 0; i < time; i++) {
            amount = amount * (1 + (rate / 100) / compoundedPerYear);
            amount = Math.pow(amount, compoundedPerYear);
        }
        return amount;
    }
}

The amount after 5 years is: 1276.28
