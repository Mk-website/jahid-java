class BPr69 {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);

    System.out.print("Enter the first string: ");
    String str1 = scanner.next();

    System.out.print("Enter the second string: ");
    String str2 = scanner.next();

  }
