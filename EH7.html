class Exception7{
public static void checkVowels(String input) throws Exception{
boolean hasVowels = false;
for (char c : input.toLowerCase().toCharArray()){
if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u'){
hasVowels = true;
break;
}
}
if (!hasVowels){
throw new Exception("String does not contain vowels : " + input);
}
System.out.println("String contains vowels : " + input);
}
public static void main(String st[]){
try{
checkVowels("Exception Handling"); 
checkVowels("rhythm"); 
}
catch(Exception e){
System.out.println(e.getMessage());
}
}
}
