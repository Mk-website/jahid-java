import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;

class EH4{
public static void readNumbersFromFile(String fileName) throws IOException, Exception {
try (BufferedReader reader = new BufferedReader(new FileReader(fileName))) {
      String line;
      while ((line = reader.readLine()) != null) {
        String[] numbers = line.split(" ");
        for (String numberStr : numbers) {
          int number = Integer.parseInt(numberStr);
          if (number > 0) {
            throw new Exception("Positive number found: " + number);
          }
          System.out.println(number);
        }}} 
    catch (IOException e) {
      throw new IOException("The file " + fileName + " was not found.", e);
    }}
  public static void main(String[] args) {
    try {
      readNumbersFromFile("numbers.txt"); 
    } catch (IOException e) {
      System.out.println(e.getMessage());
    } catch (Exception e) {
      System.out.println(e.getMessage());
    }}}
