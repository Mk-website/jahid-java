import java.util.InputMismatchException;
import java.util.Scanner;

class forshortint{
public static void main(String st[]){
Scanner scanner = new Scanner(System.in);
while (true){
try{
System.out.print("Enter the short integer value : ");
short userInput = scanner.nextShort();
System.out.println("Entered value : " + userInput);
break;
} 
catch(InputMismatchException e){
System.out.println("Invalid Value, Please Enter a valid short integer.");
scanner.next(); 
}
}
scanner.close();
 }
}
