int[][] matrix = {{1, 0, 0, 0},{0, 0, 3, 6},{4, 0, 0,9}};
int zeroCount = 0;

System.out.println("Length of column[0] : "+matrix[0].length);
int totalElements = matrix.length * matrix[0].length;
for (int i = 0 ; i < matrix.length ; i++) {
for (int j = 0 ; j < matrix[i].length; j++) {

}}}
