import java.util.InputMismatchException;
import java.util.Scanner;

class forboolean{
public static void main(String st[]){
Scanner scanner = new Scanner(System.in);
while (true){
try{
System.out.print("Enter a boolean(true or false) value : ");
boolean userInput = scanner.nextBoolean();
System.out.println("Entered value : " + userInput);
break;
} 
catch(InputMismatchException e){
System.out.println("Invalid Value, Please Enter a boolean(true or false) value.");
scanner.next(); 
}
}
scanner.close();
 }
}
