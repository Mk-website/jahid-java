class BPr94 {
    public static void main(String[] args) {
        int[] array = {12, 45, 7, 23, 56, 89, 34};
        int lowest = array[0];
        int secondLowest = array[0];

        for (int i = 1; i < array.length; i++) {
            if (array[i] < lowest) {
                secondLowest = lowest;
                lowest = array[i];
            } else if (array[i] < secondLowest && array[i] != lowest) {
                secondLowest = array[i];
            }
        }

        if (lowest == secondLowest) {
            System.out.println("There is no second lowest element");
        } else {
            System.out.println("The second lowest element is " + secondLowest);
        }
    }
}

OUTPUT



The second lowest element is 12
