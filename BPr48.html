class BPr48{
    public static void main(String[] args) {
        int hits = 100; // number of hits
        int atBats = 300; // number of at bats
        double battingAverage = calculateBattingAverage(hits, atBats);
        System.out.printf("The batting average is: %.3f%n", battingAverage);
    }
    public static double calculateBattingAverage(int hits, int atBats) {
        if (atBats == 0) {
            return 0.0; // avoid division by zero
        }
        return (double) hits / atBats;
    }
}

The batting average is: 0.333
