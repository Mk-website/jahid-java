class BPr41 {
  public static void main(String[] args) {
    double[] marks = {80, 70, 90}; 
    double sum = 0;
    for (double mark : marks) {
      sum += mark;
    }
    double average = sum / marks.length;
    System.out.println("The average mark is: " + average);
  }
}
