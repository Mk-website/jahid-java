class BPr101{
    public static void main(String[] args) {
        int[] array = {1, 2, 7, 16, 5, 6};
        for (int i = 0; i < array.length; i++) {
            for (int j = i + 1; j < array.length; j++) {
                if (array[i] < array[j]) {
                    // Swap the elements
                    int temp = array[i];
                    array[i] = array[j];
                    array[j] = temp;
                }}}
        for (int i = 0; i < array.length; i++) {
            System.out.print(array[i] + " ");
