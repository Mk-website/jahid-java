class BPr26{
    public static void main(String[] args) {
        int number = 12345; // example input

        int sum = addDigits(number);

        System.out.println("Sum of digits of " + number + ": " + sum);
    }

    public static int addDigits(int number) {
        int sum = 0;

        while (number != 0) {
            int digit = number % 10;
            sum += digit;
            number /= 10;
        }

        return sum;
    }
}

OUTPUT



Sum of digits of 12345: 15
