class Fun13{
public static boolean areAllVowels(String str) {
        str = str.toLowerCase(); // convert the string to lowercase for case-insensitive comparison
        for (char c : str.toCharArray()) {
            if (!(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')) {
                return false; // if any character is not a vowel, return false
            }
        }
        return true; // if all characters are vowels, return true
    }

    public static void main(String[] args) {
        System.out.println(areAllVowels("aeiou"));  // true
        System.out.println(areAllVowels("hello"));  // false
        System.out.println(areAllVowels("AEIOU"));  // true
    }
}
