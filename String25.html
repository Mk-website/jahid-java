class Str25 {
    public static void main(String[] args) {
        String str = "hello world hello java java world";
        printDuplicateWords(str);
    }

    public static void printDuplicateWords(String str) {
        String[] words = str.split("\\s+"); // split the string into words
        int[] wordCounts = new int[words.length];

        for (int i = 0; i < words.length; i++) {
            int count = 0;
            for (int j = 0; j < words.length; j++) {
                if (words[i].equals(words[j])) {
                    count++;
                }
            }
            wordCounts[i] = count;
        }

        System.out.println("Duplicate words: ");
        for (int i = 0; i < words.length; i++) {
            if (wordCounts[i] > 1) {
                System.out.println(words[i] + " - " + wordCounts[i] + " times");
            }
        }
    }
}

hello - 2 times
