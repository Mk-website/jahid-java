class SwapVariables{
public static void main(String st[]){
int a = 32;
int b = 77;        
System.out.println("Before swapping:");
System.out.println("a = " + a);
System.out.println("b = " + b);
int temp = a;
a = b;
b = temp;
System.out.println("After swapping:");
System.out.println("a = " + a);
System.out.println("b = " + b);
}
}

OUTPUT



Before swapping:
a = 32
b = 77
After swapping:
a = 77
b = 32
