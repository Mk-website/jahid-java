class Arr21{
public static void main(String[] args) {
int[] array = {1, 2, 2, 3, 4, 4, 5,9,9,12,13,24};
int n = array.length;
int[] temp = new int[n];
int newSize = 0;
for (int i = 0; i < n; i++){
boolean isDuplicate = false;
for (int j = 0; j < newSize; j++) {
if (array[i] == temp[j]) {
isDuplicate = true;
break;
}}
if(!isDuplicate){
temp[newSize] = array[i];
newSize++;
}}
int[] uniqueArray = new int[newSize];
for (int i = 0; i < newSize; i++) {
uniqueArray[i] = temp[i];
}
System.out.print("Array without duplicates: ");
for (int i = 0; i < uniqueArray.length; i++) {
System.out.print(uniqueArray[i] + " ");
}}}

OUTPUT



Array without duplicates: 1 2 3 4 5 9 12 13 24
