class BPr97{
    public static void main(String[] args) {
        int[] array = {0, 1, 0, 1, 0, 0, 1, 1, 0};

        int nextZero = 0;

        for (int i = 0; i < array.length; i++) {
            if (array[i] == 0) {
                // Swap array[i] with array[nextZero]
                int temp = array[nextZero];
                array[nextZero] = array[i];
                array[i] = temp;

                nextZero++;
            }
        }

        System.out.println("Segregated array:");
        for (int i = 0; i < array.length; i++) {
            System.out.print(array[i] + " ");
        }
    }
}

0 0 0 0 0 1 1 1 1
