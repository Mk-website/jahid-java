int arr[][] = {{1, 0, 0}, {0, 1, 0}, {0, 0, 1}};
boolean isIdentity = true;
for (int i = 0 ; i < arr.length ; i++) {
for (int j = 0 ; j < arr[i].length ; j++) {
if (i == j){
if (arr[i][j] != 1){
isIdentity = false;
break;
}} 
else{
if (arr[i][j] != 0) {
isIdentity = false;
break;
}}}
if (!isIdentity) {
break;
}}

}
