class BPr49{
    public static void main(String[] args) {
        double salesAmount = 1000.0; // sales amount
        double commissionRate = 0.05; // commission rate (5%)
        double commissionAmount = calculateCommission(salesAmount, commissionRate);
        System.out.printf("The commission amount is: $%.2f%n", commissionAmount);
        System.out.printf("The commission percentage is: %.2f%%%n", commissionRate * 100);
    }
    public static double calculateCommission(double salesAmount, double commissionRate) {
        return salesAmount * commissionRate;
    }
}
